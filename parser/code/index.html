Code

The code are mainly the two files htmlToLatex.js and domToLatex.js. They depend on htmlparser2 and latex-sanitizer to build a compilable .tex document form any given html source code.

htmlToLatex.js

htmlToLatex.js sets up the htmlparser2 and uses the generated dom to convert that to latex via domToLatex.js

Note

At this point htmlToLatex.js only supports htmlparser2's parseComplete function.

var htmlparser = require('htmlparser2');
var domToLatex = require('./lib/domToLatex');

/**
 * wrapper for
 * @param  {function name} parserCall         function to call on the parser
 * @param  {object}        parserOptions      options to call the parserCall with
 * @param  {string}        options.latexType  given latexType to use to parse      
 * @return {string}                           string with built latex code
 */
module.exports = function(parserCall, parserOptions, options) {

  var parsedLatex = '';

  // define dom handler for htmlparser2
  // the handler get called after the html code was parsed to a dom object
  var handler = new htmlparser.DomHandler(function (error, dom) {
    if (!error) {
      parsedLatex = '';

      // convert every dom element its latex equivalent with domToLatex
      for (var i = 0; i < dom.length; i++) {
        // see domToLatex.js for that function
        parsedLatex += domToLatex(dom[i], options.latexType);
      }
      return parsedLatex;
    }
  });

  // create htmlparser with handler
  var htmlToLatex = new htmlparser.Parser(handler);

  // take the parserCall
  switch (parserCall) {

    // only parseComplete is needed for now
    case 'parseComplete': 
      if(parserOptions.length === 1) {
        // use parseComplete
        htmlToLatex.parseComplete(parserOptions[0]);
      } else {
        return null;
      }
    break;
  }
  return parsedLatex;
}

domToLatex.js

The domToLatex function recursivly converts a dom element to latex code by using given templates. Every text (that is not part of the markup) is escaped and sanitized using latex-sanitizer.

// load the latex-sanitizer
var sanitizeLatex    = require('latex-sanitizer');

// load the templates
var templates        = {};
templates['article'] = require("dot").process({path: "./parser/templates/defaultArticle"});
templates['letter']  = require("dot").process({path: "./parser/templates/defaultLetter"});

// load the configurations
var config           = {};
config['article']    = require('./articleConfig');
config['letter']     = require('./letterConfig');

/**
 * Looks if given style is part of that classtype
 * @param  {string} inlineStyle inline CSS string
 * @param  {string} tag_name    tag that style belongs to
 * @param  {string} classType   type of document i.e. article
 * @return {[string]}           list of template names to use
 */
var parseStyles = function(inlineStyle, tag_name, classType){

  // Parses styles like text-align: center to 
  //    style[1] = text-align
  //    style[2] = center
  var CSS_REGEX = /([\w-]+)\s*:\s*([^;]+)\s*;?/g;
  var style;

  var styleTemplates = [];
  while ((style = CSS_REGEX.exec(inlineStyle)) !== null) {
    // check if found style is allowed and defined for this tag
    if (typeof config[classType]['allowedStyles'][style[1]] !== 'undefined' && config[classType]['allowedStyles'][style[1]].indexOf(tag_name) !== -1) {
      // check if there is a template for that style option
      if (typeof config[classType]['styles'][style[1]][style[2]] !== 'undefined') {
        styleTemplates.push(config[classType]['styles'][style[1]][style[2]]);
      }
    }
  }

  return styleTemplates;
}

/**
 * Semi-recursivly parses the htmlparser2-domTree by searching for matches between tags and templates
 * @param  {json}   node      the node with all it's info and children
 * @param  {String} classType option what template set-up to use
 * @return {String}           A latex string all parsed and escaped
 */
var domToLatex = function(node, classType){
  // make sure classType is set
  classType = classType || 'article';

  // end of recursion when there is just text
  if (node.type === 'text') {
    return sanitizeLatex(node.data);
  }

  // if a tag is found
  if (node.type === 'tag') {
    // get the tag_name to use
    var tag_name = config[classType].tags[node.name] || node.name;

    // convert all children to latex first
    var text = '';
    for (var i = 0; i < node.children.length; i++) {
      text += domToLatex(node.children[i], classType);
    };

    // add style templates if necessary
    if(typeof node.attribs.style !== 'undefined'){
      // call above defined function
      var styleTemplates = parseStyles(node.attribs.style, tag_name, classType);

      // use every returned template name (if possible)
      for (var i = 0; i < styleTemplates.length; i++) {
        if(typeof templates[classType][styleTemplates[i]] !== 'undefined'){
          text = templates[classType][styleTemplates[i]]({"content": text});
        }
      };
    }

    // use tag templates (if possible)
    if(typeof templates[classType][tag_name] !== 'undefined'){
      return templates[classType][tag_name]({"content": text});
    } else {
      return text;
    }
  }
}
