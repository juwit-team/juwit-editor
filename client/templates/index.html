The starting point for the frontend is the /client/templates/index.html it includes all styles and javascript dependencies, contains the main navigation and the ngView.

<!DOCTYPE html>
<html>
<head>
  <!-- styles -->
</head>
<body ng-app="LatexEditor">

  <!-- static navigation -->
  <nav class="navbar navbar-inverse navbar-fixed-top">
    <!-- .. -->
  </nav>

  <!-- entrypont for ngRoutes: -->
  <div ng-view></div>

  <!-- JS dependencies -->
  <script src='/bower_components/angular/angular.js'></script>
  <!-- .. -->


</body>
</html>

Editor

The only available route at this point is the editor. When the server is running http://localhost:1338/#editor will render /client/templates/editor/index.html between the ngView tag. This template adds the textAngular-editor and the pdf-preview to the user view.

<div class="app">

  <!-- second navbar right under the static top bar -->
  <nav class="navbar subnav">
    <div class="container">
      <div class="row">
        <!-- toolbar for article -->
        <div text-angular-toolbar ... ></div> 
        <!-- toolbar for letter -->
        <div text-angular-toolbar ... ></div> 

        <!-- switch article to letter or even form letter -->
        <select class="form-control" ng-model="selectedTemplate" ng-options="template.name for template in templates" ng-change="modal()">
          </select> 

        <!-- edit options for selected template -->
        <button name="editForm" data-target="{{selectedTemplate.modaltarget}}" ng-show="selectedTemplate.editable">Edit</button>

        <!-- download button to save the pdf -->
        <button name="downloadButton" ng-click="download()">Download</button>
      </div>
    </div>
  </nav>

  <!-- editor and preview -->
  <div class="container">  
    <div class="row">
      <div class="col-sm-7">
        <!-- textangular area -->
        <div class="wysiwyg" text-angular="text-angular" name="htmlcontent" ng-model="data.htmlContent" ta-target-toolbars='statictoolbarArticle,statictoolbarFormLetter'>
        </div>
      </div>
      <div class="col-sm-5">
        <!-- preview !with hard coded default! -->
        <pdf-preview url="/dl/company/document/document.pdf"></pdf-preview>
    </div>  
  </div>
</div>

To see how the editor view interacts with the server look at the editor/index-controller
