Routes

The routes get called when a matching request is send to the server. The routes are there to edit the input for the matching model. We chose a semantic URL-scheme where you first go to the document you want to edit and then call functions on that document.

compile.js

The compile route only listens for post requests to compile the sent data.

URL-scheme: http://localhost:1338/{group_name}/{document_title}/compile

Note

Because of the scrum development process and a few changed design decisions the parser gets called in the route for now. Also the use of templates is not done/finished yet.

// require the models and the parser (simplified)
var Compiler, FormletterCompiler, Parser, latexTemplates = require('...');

module.exports = function(app) {

  // listen for URL-scheme `http://localhost:1338/{group_name}/{document_title}/compile`
  app.post('/:group/:document/compile', function(request, response){
    // set local variables from request (simplified)
    var filename, group, latexType, html, latexCode = ...;

    // FormletterCompiler works differently
    if(latexType === 'bulk-letter') {   
      FormletterCompiler.compile(request);
    } else {
      // if it's a normal artice or letter parse the given latex
      var parsedLatex = Parser('parseComplete', [html], {latexType: latexType});

      // set the latex head according to the document type
      if (latexType === 'letter') {
        // generate from letter template
        latexCode = latexTemplates.letter({sender: request.body.sender, recipient: request.body.recipient, latexCode: parsedLatex});
      } else  {
        // everything other than bulk-letter and letter are handled as article
        latexType = 'article';
        // generate from article template
        latexCode = latexTemplates.article({latexCode: parsedLatex});
      }

      // compile the generated latex code
      Compiler.compile(group, filename, latexCode, function (jsonResponse) {
        response.json(jsonResponse);
      });
    }
  });
};

updateCSV.js

This route allows CSV file uploads for the CSV that is used for the formletter. It only listen to post requests to the following URL-scheme.

URL-scheme: http://localhost:1338/{group_name}/{document_title}/updateCSV

// require the filesystem and exec command (simplified)
var fs, exec = require('...');

module.exports = function(app) {

  // listen for URL-scheme `http://localhost:1338/{group_name}/{document_title}/updateCSV`
  app.post('/:group/:document/updateCSV', function(request, response) {

    // set local variables from request (simplified)
    var latexDocumentName, group, src, dst = ...;

    // move file to corresponding document folder
    var mvFileCmd = 'mkdir -p '+dst+'; mv '+src+' '+dst+'/'+request.files.file.name;
    exec(mvFileCmd);

    // answer with success
    response.json({"status": "File uploaded."});
  });
};
