compiler.js

The compiler only exports the compile() function that uses the pdflatex wrapper to generate the pdf at a given path.

/**
 * takes latexCode and produces the pdf at given path
 * @param  {string} group_name     name of company/group
 * @param  {string} document_title title of document
 * @param  {string} latexCode      latex code as a string
 * @param  {function} _callback    function that gets called at the end
 * @return _callback is called at end with a return object (non blocking)       
 */
function compile(group_name, document_title, latexCode, _callback) {

  // paths
  var fileInDirectory, outputDirectory, defaultPath;

  // copy class files from company/group templates
  var shellCmd; exec(shellCmd);

  // save latexCode and then call the function
  fs.writeFile(fileInDirectory, latexCode, function(err){

    // call pdflatex-wrapper
    var tex = new pdflatex(fileInDirectory, outputDirectory, doc);
    tex.compile(function (jsonResponse) {

      if(jsonResponse.error) {
        // if the compile fails: call _callback with error object
        _callback({error: jsonResponse.error});
      } else {
        // if everything is fine call _callback with path to download pdf
        // Math.random() generates a random number as cache buster
        _callback({redirect: '/dl/' + group_name + '/' + document_title + '/' + document_title + '.pdf?v=' + Math.random() });
      };
    });
  });
}

pdflatex.js

A thin wrapper for the pdflatex command based on node-pdflatex. There where issues with the command itself. It was not working properly on windows and the interaction nonstopmode was missing. Also 

// construktor
function PDFlatex(inputPath, outputDirectory, filename) {
  // default settings with fix for windows path names
  this.outputDirectory = outputDirectory.replace(/\\/g, '/');
  this.inputPath = inputPath.replace(/\\/g, '/');
  this.filename = filename;
};

/**
 * calls pdflatex and saves the pdf with given path parameters
 * @param  {function} _callback function that gets called after pdflatex is done
 * @return calls _callback with an return object to work with (non-blocking)
 */
PDFlatex.prototype.compile = function(_callback) {
  if (this.inputPath && this.inputPath.length > 0) {
    // build command
    var command = "pdflatex -interaction nonstopmode -output-directory " + this.outputDirectory + " " + this.inputPath;

    // execute command
    var parent = this;
    exec(command, function(error, stdout, stderr) {
      if (error !== null) {
        // if pdflatex fails call _callback with an error
        _callback({error: 'Could not create ' + parent.filename + '.pdf'});
      } else {
        // if pdflatex succeds call _callback with the new path to pdf
        _callback({filePath: __server + '_texFiles/' + parent.filename +'.pdf'});
      }
    });
  }
};

formletter-compiler.js

The idea is that the CSV contains all the relevant data for every recipient. Then for every recipient the latexCode gets modified (for the letter head) and then compiled.

Warning

This part is very quick and dirty and therefore unstable

/**
 * compiles the latex from html and generates multiple letters
 * @param  {string} request.params.document document title
 * @param  {string} request.params.group    group/company name
 * @param  {string} request.body.htmlCode   unparsed htmlCode
 */
function compile(request) {
  // set the local variables (simplified)
  var html = request.body.htmlCode;
  var filename, group, html, csvFilepath;

  // read the csv and do something with the read data
  var csvParser = CSVParse({delimiter: ','}, function(err, csvData) {

    // loop through every entry 
    for(recordIndex = 1; recordIndex < csvData.length; recordIndex++) {

      // to avoid side effects a semaphore is taken
      Semaphore.take(function() {
        // call the parser
        var parsedLatex =  Parser('parseComplete', [html], {latexType: 'letter'});

        // build latexCode for this specific record (simplified)
        latexCode  = '\\documentclass{defaultLetter}';
        // latexCode += ...
        latexCode += parsedLatex;
        // latexCode += ..
        latexCode += '\\end{document}';

        // compile newly built latex
        Compiler.compile(group, filename, latexCode, function (jsonResponse) {

          // move the pdf into the folder that gets zipped at the end
          var shellCmd = 'mkdir -p ' + documentDirectory + 'zipme/' + '; '
                  + 'mv ' + documentDirectory + filename + '.pdf '
                  + documentDirectory + 'zipme/' + filename + Date.now() + '.pdf';

          exec(shellCmd);

          Semaphore.leave();
          });
        }
      }
    }

    // zip the newly created folder
    Semaphore.take(function() {
      Zipper.zippdfs(documentDirectory + 'zipme/', documentDirectory);
      Semaphore.leave();
    });
  }
}
